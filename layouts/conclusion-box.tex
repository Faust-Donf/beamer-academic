% Layout: conclusion-box (结论框页)
% 使用场景: 章/分析结束时总结核心发现，用高亮框突出关键结论
%
% Slots:
%   {{TITLE}}      - 页标题
%   {{CHAPNOTE}}   - 对应论文章节标注（可选）
%   {{BODY_TEXT}}  - 总结性正文（100-150字）
%   {{KEYBOX}}     - 高亮框内容（核心结论，50-80字）
%
% 视觉效果: 正文在上，keybox 在下方用浅灰底+细线框突出

\begin{frame}
  \frametitle{{{TITLE}}}
  {{CHAPNOTE_LINE}}
  \vskip0.2cm

  {{BODY_TEXT}}

  \vskip0.3cm

  \keybox{{{KEYBOX}}}
\end{frame}
